Python Example On Exception Handling (1/2)

def divide_traditional(dividend, divisor):
    if divisor == 0:
        return None # Signal error/failure
    else:
        return dividend / divisor

# Caller
result = divide_traditional(10, 0)
if result is None:
    print("Error: Division by zero detected (traditional)")
else:
    print(f"Success: Result is {result}")

Python Example On Exception Handling (2/2)

def divide_exceptions(dividend, divisor):
    # No explicit check needed here for ZeroDivisionError
    # Python raises it automatically if divisor is 0
    return dividend / divisor

# Caller
try:
    result = divide_exceptions(10, 0)
    print(f"Success: Result is {result}") # Won't be reached
except ZeroDivisionError:
    print("Error: Caught ZeroDivisionError exception")
